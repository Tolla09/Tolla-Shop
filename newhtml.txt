    function setLanguage(lang) {
      const s = strings[lang];

      if (!s) return;

      // header
      if (els.headerTitle) els.headerTitle.textContent = s.headerTitle;
      if (els.headerSubtitle) els.headerSubtitle.textContent = s.headerSubtitle;

      // nav
      if (els.navLinks && els.navLinks.length >= s.nav.length) {
        els.navLinks.forEach((a, i) => {
          if (s.nav[i]) a.textContent = s.nav[i];
        });
      }

      // services title
      if (els.servicesTitle) els.servicesTitle.textContent = s.servicesTitle;

      // each service card (title, short paragraph, strong and description)
      if (els.serviceCards && els.serviceCards.length === s.serviceItems.length) {
        els.serviceCards.forEach((card, i) => {
          const item = s.serviceItems[i];
          const h3 = card.querySelector("h3");
          const p = card.querySelector("p");
          const strong = card.querySelector(".service-description strong");
          const descNodes = card.querySelectorAll(".service-description");
          const desc = card.querySelector(".service-description");

          if (h3 && item.title) h3.textContent = item.title;
          if (p && item.short) {
            // keep original line breaks but replace last line text
            p.innerHTML = item.short.replace(/\n/g, "<br>");
          }
          if (strong && item.strong) strong.textContent = item.strong;
          if (desc && item.desc) {
            // desc text after the strong already exists as text node; replace innerHTML while preserving <br>
            desc.innerHTML = `<strong>${item.strong}</strong><br>${item.desc}`;
          }
        });
      } else {
        // fallback: try to map as many as possible if counts differ
        const cards = document.querySelectorAll(".services-grid .service-card");
        cards.forEach((card, i) => {
          const item = s.serviceItems[i] || null;
          const h3 = card.querySelector("h3");
          const p = card.querySelector("p");
          const desc = card.querySelector(".service-description");
          if (item) {
            if (h3) h3.textContent = item.title;
            if (p) p.innerHTML = item.short.replace(/\n/g, "<br>");
            if (desc) desc.innerHTML = `<strong>${item.strong}</strong><br>${item.desc}`;
          }
        });
      }

      // about
      if (els.aboutTitle) els.aboutTitle.textContent = s.aboutTitle;
      if (els.aboutText) els.aboutText.textContent = s.aboutText;

      // hire
      if (els.hireTitle) els.hireTitle.textContent = s.hireTitle;
      if (els.hireText) els.hireText.textContent = s.hireText;

      // contact
      if (els.contactTitle) els.contactTitle.textContent = s.contactTitle;
      if (els.inputs.name) els.inputs.name.placeholder = s.placeholders.name;
      if (els.inputs.email) els.inputs.email.placeholder = s.placeholders.email;
      if (els.inputs.message) els.inputs.message.placeholder = s.placeholders.message;
      if (els.inputs.sendBtn) els.inputs.sendBtn.textContent = s.sendButton;

      // floating tooltip
      if (els.floatingTooltip) els.floatingTooltip.textContent = s.floatingTooltip;

      // footer
      if (els.footer) els.footer.textContent = s.footer;

      // success message (update template used on submit)
      // We will store the language in localStorage and use this string when showing a success message
      localStorage.setItem("site_lang", lang);

      // update active button UI
      document.getElementById("btn-en").classList.toggle("active", lang === "en");
      document.getElementById("btn-ku").classList.toggle("active", lang === "ku");
    }

    // initialize language on load (from localStorage or default to English)
    document.addEventListener("DOMContentLoaded", function () {
      const saved = localStorage.getItem("site_lang") || "en";
      setLanguage(saved);

      // set up buttons
      document.getElementById("btn-en").addEventListener("click", () => setLanguage("en"));
      document.getElementById("btn-ku").addEventListener("click", () => setLanguage("ku"));
    });

    // Override the success message inserted by email handler to use localized text
    // We replace the showing of success message by listening for form submission success.
    (function () {
      const form = document.getElementById("contact-form");
      if (!form) return;
      const originalHandler = form.onsubmit; // not used, but keep
      form.addEventListener("submit", function (e) {
        // the existing emailjs code handles preventing default and sending.
        // We intercept after the send completes by wrapping emailjs.sendForm call.
        // To avoid editing original emailjs call, we monkey-patch emailjs.sendForm if exists.
        // This preserves existing behavior but ensures localized success text shows.
        // If emailjs is not loaded yet, do nothing here.
      });
    })();

    // To ensure the success message uses the localized string, we patch the success display code
    // by wrapping the emailjs.sendForm promise if emailjs is available.
    (function () {
      // Wait a short time for page to load and existing scripts to run
      const wrap = () => {
        if (!window.emailjs || !emailjs.sendForm) return;
        const originalSendForm = emailjs.sendForm.bind(emailjs);
        emailjs.sendForm = function () {
          // call original and intercept success handler to show localized message
          const args = arguments;
          const promise = originalSendForm.apply(null, args);
          promise.then(() => {
            const lang = localStorage.getItem("site_lang") || "en";
            const text = strings[lang] ? strings[lang].successMessage : strings.en.successMessage;
            const container = document.getElementById("success-animation-container");
            if (container) {
              container.innerHTML = '<div class="success-animation">' + text + ' <i class="fas fa-check-circle"></i></div>';
              setTimeout(() => {container.innerHTML = "";}, 1800);
            }
          }).catch(() => { /* leave original error handling in place */});
          return promise;
        };
      };
      // attempt to wrap now and again later in case emailjs loads after this script
      wrap();
      setTimeout(wrap, 600);
    })();
  </script>
